331  one_hot_vec[i].resize(unique_category_vector.size());

333 
334  // label row 0 of one_hot_vec with the new column names
335  std::string column_name = data_vec[0][categoricalIndex];
336  std::unordered_map<std::string, int> one_hot_col_index; // one_hot_col_index["CategoryName"] = column_index
337  for(size_t i = 0; i < unique_category_vector.size(); i++) {
338  std::string category_name = unique_category_vector[i];
339  std::string one_hot_col_name = column_name + ":" + category_name;

341  one_hot_col_index[category_name] = i;
342  }
343

346  std::string category_name = data_vec[row][categoricalIndex]; // retrieve the category name from the row
347  int index = one_hot_col_index[category_name]; // retrieve the column index in our one hot vector for the category name
348  for(int col = 0; col < unique_category_vector.size(); col++) {
349  // if we are at the correct index, put 1, else 0
