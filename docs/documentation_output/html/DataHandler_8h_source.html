169  std::vector<std::vector<std::string>> impute_data(std::vector<std::vector<std::string>> data_vec, std::vector<int> categorical_indexes) {
170  for(size_t col = 0; col < data_vec[0].size(); col++) {
171  auto it = std::find(categorical_indexes.begin(), categorical_indexes.end(), static_cast<int>(col));
172  if(it != categorical_indexes.end()) { // is categorical
173  /*
174  declare unordered_map to count occurrence, impute missing data with most occurring category
175  e.g: occurrence_map["Yes"] = 123, occurrence_map["No"] = 432 || Missing values will be replaced with "No" since it occured more often.
176  */
177  std::unordered_map<std::string, int> occurrence_map;
178 
179  // count occurrences
180  for(size_t row = 1; row < data_vec.size(); row++) {
181  std::string cell = data_vec[row][col];
182  if(cell == "NULL") continue;
183  occurrence_map[cell]++;
184  }
185 
186  // find the most occurring category
187  std::string most_occurring_category = most_occurred_key_in_map(occurrence_map);
188 
189  // replace NULL values
190  for(size_t row = 1; row < data_vec.size(); row++) {
191  if(data_vec[row][col] != "NULL") continue;
192  data_vec[row][col] = most_occurring_category;
193  }
194 
195  } else { // not categorical
196  // Replace non-categorical NULL values with the mean
197 
198  double sum = 0;
199  size_t count = 0;
200  // calculate mean
201  for(size_t row = 1; row < data_vec.size(); row++) {
202  if(data_vec[row][col] == "NULL") continue;
203  try {
204  double cell = std::stod(data_vec[row][col]);
205  sum += cell;
206  count++;
207  } catch(...) {
208  continue;
209  }
210  }
211  // replace NULL values
212  for(size_t row = 1; row < data_vec.size(); row++) {
213  if(data_vec[row][col] != "NULL") continue;
214  if(!count) { // avoid division by 0
215  data_vec[row][col] = "0";
